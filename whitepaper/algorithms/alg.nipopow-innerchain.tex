\begin{algorithm}[H]
    \caption{\label{alg.nipopow-innerchain}constructInnerChain}
    \begin{algorithmic}[1]
        \Function{\sf constructInnerChain}{$\chain, \mu, b$}
            \Let{B}{\chain[-1]}
            \Let{\textsf{innerChain}}{[B]}
            \While{$B \neq b$}
                \Let{id}{B.\textsf{interlink}[\mu]}
                \Let{B}{\textsf{blockById}[id]}
                \State{\textsf{innerChain.append}(B)}
            \EndWhile
            \State\Return{\textsf{innerChain}}
        \EndFunction
    \vskip2pt
    \end{algorithmic}
\end{algorithm}
