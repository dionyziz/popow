\begin{algorithm}[H]
    \caption{\label{alg.nipopow-infix-follow}The followDown function which
    produces the necessary blocks to connect a superblock $hi$ to a preceeding
    regular block $lo$.}
    \begin{algorithmic}[1]
        \Function{\sf followDown}{$hi$, $lo$, height} %, realLink, blockById}
            \State{$B \gets hi$; $aux \gets [\ ]$; $\mu \gets \textit{level}(hi)$}
            \While{$B \neq lo$}
                % \Let{(B', aux')}{\textsf{followUp}(B, \mu, \textsf{realLink}, \textsf{blockById})}
                \Let{B'}{\textsf{blockById}[B\text{.interlink}[\mu]]}
                \If{$\textsf{height}[B'] < \textsf{height}[lo]$}
                    \Let{\mu}{\mu - 1}
                \Else
                    \Let{aux}{aux \cup \{B\}}
                    \Let{B}{B'}
                \EndIf
            \EndWhile
            \State\Return{$aux$}
        \EndFunction
    \vskip8pt
    \end{algorithmic}
\end{algorithm}
